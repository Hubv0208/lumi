async function fetchWeather() {

async (pos) => {
const {latitude:lat,longitude:lon}=pos.coords;
btn.textContent=‘🌤 Fetching weather…’;
setWeatherStatus(‘Got location, fetching weather…’);
try {
const url=`https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&current=temperature_2m,apparent_temperature,relative_humidity_2m,wind_speed_10m,weather_code&wind_speed_unit=kmh&timezone=auto`;

const cur=data.current;
lastWeatherC=cur.temperature_2m;
lastWeatherCond=wmoToCondition(cur.weather_code);
cachedWeather={
tempC: cur.temperature_2m,
feelsC: cur.apparent_temperature,
humidity: cur.relative_humidity_2m,
windKmh: cur.wind_speed_10m,
condition: lastWeatherCond,
desc: wmoToDesc(cur.weather_code),
lat, lon
};
// Reverse geocode with Open-Meteo doesn’t give city — use a free endpoint
let cityName=‘Your location’;
try {
const geo=await fetch(`https://nominatim.openstreetmap.org/reverse?lat=${lat}&lon=${lon}&format=json`);
const gd=await geo.json();
cityName=gd.address.city||gd.address.town||gd.address.village||gd.address.county||‘Your location’;
} catch(e) {}
cachedWeather.city=cityName;
updateWeatherDisplay();
setWeatherStatus(’’,false);

renderRecommendation(lastWeatherC,lastWeatherCond,tod);
} catch(e) {

btn.disabled=false; btn.textContent=‘📍 Get current weather’;
},
(err) => {

);

// WMO weather code → condition key
function wmoToCondition(code) {
if (code===0||code===1) return ‘clear’;
if (code===2||code===3) return ‘cloudy’;
if (code>=45&&code<=48) return ‘fog’;
if ((code>=51&&code<=67)||(code>=80&&code<=82)) return ‘rain’;
if ((code>=71&&code<=77)||(code>=85&&code<=86)) return ‘snow’;
if (code>=95) return ‘rain’;
return ‘cloudy’;
}
function wmoToDesc(code) {
if (code===0) return ‘Clear sky’;
if (code===1) return ‘Mainly clear’;
if (code===2) return ‘Partly cloudy’;
if (code===3) return ‘Overcast’;
if (code>=45&&code<=48) return ‘Foggy’;
if (code>=51&&code<=55) return ‘Drizzle’;
if (code>=61&&code<=65) return ‘Rain’;
if (code>=71&&code<=75) return ‘Snow’;
if (code>=80&&code<=82) return ‘Rain showers’;
if (code>=85&&code<=86) return ‘Snow showers’;
if (code>=95) return ‘Thunderstorm’;
return ‘Mixed conditions’;
}

// Returns array of {label, leftMins, rightMins, leftCount, rightCount} one per day
function buildDayBuckets(days) {
const buckets = [];
const now = new Date();
for (let i = days-1; i >= 0; i–) {
const d = new Date(now);
d.setDate(d.getDate() - i);
const key = d.toDateString();
const label = i === 0 ? ‘Today’ :
i === 1 ? ‘Yest.’ :
d.toLocaleDateString([],{weekday:‘short’});
const dayFeeds = feeds.filter(f => new Date(f.id).toDateString() === key);
const lf = dayFeeds.filter(f => f.side === ‘L’);
const rf = dayFeeds.filter(f => f.side === ‘R’);
buckets.push({
label,
leftMins:  lf.reduce((a,b) => a + b.duration, 0) / 60,
rightMins: rf.reduce((a,b) => a + b.duration, 0) / 60,
leftCount:  lf.length,
rightCount: rf.length,
});
}
return buckets;
}

function getCanvasCtx(id) {
const canvas = document.getElementById(id);
const dpr = window.devicePixelRatio || 1;
const rect = canvas.parentElement.getBoundingClientRect();
const cssW = rect.width - 24; // account for padding
canvas.style.width  = cssW + ‘px’;
canvas.width  = cssW * dpr;
canvas.height = parseInt(canvas.getAttribute(‘height’)) * dpr;
canvas.style.height = canvas.getAttribute(‘height’) + ‘px’;
const ctx = canvas.getContext(‘2d’);
ctx.scale(dpr, dpr);
return { ctx, w: cssW, h: parseInt(canvas.getAttribute(‘height’)) };
}

function drawTimeline(canvasId) {
const {ctx, w, h} = getCanvasCtx(canvasId);
const PINK = ‘#C97B8A’, BLUE = ‘#7B9CC9’;
const today = new Date().toDateString();
const todayFeeds = feeds.filter(f => new Date(f.id).toDateString() === today);

ctx.clearRect(0, 0, w, h);

const padL = 20, padR = 20, midY = h / 2;
const trackW = w - padL - padR;

// Track line
ctx.beginPath();
ctx.strokeStyle = ‘rgba(255,255,255,0.08)’;
ctx.lineWidth = 2;
ctx.moveTo(padL, midY); ctx.lineTo(w - padR, midY);
ctx.stroke();

// Hour ticks 0–24
ctx.fillStyle = ‘rgba(255,255,255,0.2)’;
ctx.font = ‘8px -apple-system,sans-serif’;
ctx.textAlign = ‘center’;
[0, 6, 12, 18, 24].forEach(hr => {
const x = padL + (hr / 24) * trackW;
ctx.beginPath();
ctx.strokeStyle = ‘rgba(255,255,255,0.12)’;
ctx.lineWidth = 1;
ctx.moveTo(x, midY - 6); ctx.lineTo(x, midY + 6);
ctx.stroke();
ctx.fillText(hr === 24 ? ‘’ : hr+‘h’, x, h - 4);
});

if (!todayFeeds.length) {
ctx.fillStyle = ‘rgba(255,255,255,0.2)’;
ctx.font = ‘11px -apple-system,sans-serif’;
ctx.textAlign = ‘center’;
ctx.fillText(‘No feeds recorded today’, w/2, midY - 10);
return;
}

// Max duration for dot sizing
const maxDur = Math.max(…todayFeeds.map(f => f.duration), 60);

todayFeeds.forEach(f => {
const d = new Date(f.startTime);
const fracOfDay = (d.getHours() * 60 + d.getMinutes()) / (24 * 60);

const r = 5 + (f.duration / maxDur) * 12; // 5–17px radius
const color = f.side === ‘L’ ? PINK : BLUE;

```

  ctx.textAlign = 'center';
  ctx.fillText(Math.round(f.duration/60)+'m', x, midY + 4);
```

});
}

function roundRect(ctx, x, y, w, h, r) {
ctx.beginPath();
ctx.moveTo(x + r, y);
ctx.lineTo(x + w - r, y);
ctx.quadraticCurveTo(x + w, y, x + w, y + r);
ctx.lineTo(x + w, y + h);
ctx.lineTo(x, y + h);
ctx.lineTo(x, y + r);
ctx.quadraticCurveTo(x, y, x + r, y);
ctx.closePath();
}
